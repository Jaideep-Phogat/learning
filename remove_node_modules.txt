#!/bin/bash

# Function to print usage
usage() {
    echo
    echo "This script recursively removes all 'node_modules' directories within>
    echo
    echo "Usage: $0 -p <path>"
    echo
    echo "Options:"
    echo "  -p <path>   Specify the parent directory path"
    echo "  --help      Instruction regarding the tool usage"
    exit 1
}

# Parse command line arguments
while [[ "$1" != "" ]]; do
    case $1 in
        -p | --path )          shift
                               path=$1
                               ;;
        --help )               usage
                               ;;
        * )                    usage
                               ;;
    esac
    shift
done

# Check if path is provided
if [ -z "$path" ]; then
    echo "Error: Path is required."
    usage
fi

# Check if the provided path is a directory
if [ ! -d "$path" ]; then
    echo "Error: Provided path is not a directory."
    usage
fi

# Remove all node_modules directories recursively
find "$path" -name "node_modules" -type d -prune -exec rm -rf '{}' +

echo "All node_modules directories removed from $path and its subdirectories."
